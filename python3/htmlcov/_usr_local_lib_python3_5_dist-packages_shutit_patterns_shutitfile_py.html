127# Parses the shutitfile (passed in as a string) 

128# and info to extract, and returns a list with the information in a more canonical form, still ordered. 

129def parse_shutitfile(contents): 

130 ret = [] 

131 full_line = '' 

132 for line in contents.split('\n'): 

133 line = line.strip() 

134 # Handle continuations 

135 if line: 

136 if line[-1] == '\\': 

137 full_line += line[0:-1] 

138 continue 

139 else: 

140 full_line += line 

141 if re.match(r"^IF_NOT+[\s]+([A-Z_]+)[\s]+(.*)$", full_line): 

142 m = re.match(r"^IF_NOT+[\s]+([A-Z_]+)[\s]+(.*)$", full_line) 

143 ret.append(['IF_NOT',m.group(1),m.group(2)]) 

144 elif re.match(r"^STORE_RUN+[\s]+([a-zA-Z0-9_]+)[\s]+(.*)$", full_line): 

145 m = re.match(r"^STORE_RUN+[\s]+([a-zA-Z0-9_]+)[\s]+(.*)$", full_line) 

146 ret.append(['STORE_RUN',m.group(1),m.group(2)]) 

147 elif re.match(r"^ELIF_NOT[\s]+([A-Z_]+)[\s]+(.*)$", full_line): 

148 m = re.match(r"^ELIF_NOT[\s]+([A-Z_]+)[\s]+(.*)$", full_line) 

149 ret.append(['ELIF_NOT',m.group(1),m.group(2)]) 

150 elif re.match(r"^IF[\s]+([A-Z_]+)[\s]+(.*)$", full_line): 

151 m = re.match(r"^IF[\s]+([A-Z_]+)[\s]+(.*)$", full_line) 

152 ret.append(['IF',m.group(1),m.group(2)]) 

153 elif re.match(r"^ELIF[\s]+([A-Z_]+)[\s]+(.*)$", full_line): 

154 m = re.match(r"^ELIF[\s]+([A-Z_]+)[\s]+(.*)$", full_line) 

155 ret.append(['ELIF',m.group(1),m.group(2)]) 

156 elif re.match("^ELSE$", full_line): 

157 ret.append(['ELSE']) 

158 elif re.match("^ENDIF$", full_line): 

159 ret.append(['ENDIF']) 

160 elif re.match(r"^([A-Za-z_]+)[\s]*(.*)$", full_line): 

161 m = re.match(r"^[\s]*([A-Za-z_]+)[\s]*(.*)$", full_line) 

162 ret.append([m.group(1), m.group(2)]) 

163 elif re.match("^#(.*)$", full_line): 

164 # Comments should be added with 'COMMENT a comment' 

165 pass 

166 else: 

167 full_line_split = ''.join((full_line[:10000].split())) 

168 full_line_strings = re.findall("[^\x00-\x1F\x7F-\xFF]", full_line_split) 

169 shutit_global.shutit_global_object.shutit_print('FAILED TO PARSE: ' + full_line_strings[:30] + '[...]') 

170 return [], False 

171 full_line = '' 

172 return ret, True

175def parse_shutitfile_args(args_str): 

176 """Parse shutitfile args (eg in the line 'RUN some args', the passed-in args_str would be 'some args'). 

177 If the string is bounded by square brackets, then it's treated in the form: ['arg1','arg2'], and the returned list looks the same. 

178 If the string composed entirely of name-value pairs (eg RUN a=b c=d) then it's returned as a dict (eg {'a':'b','c':'d'}). 

179 If what's passed-in is of the form: "COMMAND ['a=b','c=d']" then a dict is also returned.' 

180 Also eg: ["asd and space=value","asd 2=asdgasdg"]""" 

181 ret = [] 

182 if args_str == '': 

183 return ret 

184 if args_str[0] == '[' and args_str[-1] == ']': 

185 ret = eval(args_str) 

186 assert isinstance(ret, list) 

187 else: 

188 ret = args_str.split() 

189 # if all the items have a = in them, then return a dict of nv pairs 

190 nv_pairs = True 

191 for item in ret: 

192 if item.find('=') < 0: 

193 nv_pairs = False 

194 if nv_pairs: 

195 d = {} 

196 for item in ret: 

197 item_nv = item.split('=') 

198 d.update({item_nv[0]:item_nv[1]}) 

199 ret = d 

200 return ret

271def check_shutitfile_representation(shutit, shutitfile_representation, skel_delivery): 

272 # delivery directives 

273 # Only allow one type of delivery 

274 shutitfile_delivery = set() 

275 # If we've been given a delivery method, add that. 

276 if skel_delivery: 

277 shutitfile_delivery.add(skel_delivery) 

278 for item in shutitfile_representation['shutitfile']['delivery']: 

279 shutitfile_delivery.add(item[1]) 

280 if len(shutitfile_delivery) > 1: 

281 shutit.fail('Conflicting delivery methods in ShutItFile') 

282 elif len(shutitfile_delivery) == 1: 

283 skel_delivery = shutitfile_delivery.pop() 

284 else: 

285 # Default skel_delivery to bash if none seen 

286 skel_delivery = 'bash' 

287 

288 if skel_delivery not in shutit_global.shutit_global_object.allowed_delivery_methods: 

289 shutit.fail('Disallowed delivery method in ShutItFile: ' + skel_delivery) 

290 

291 if skel_delivery != 'docker': 

292 # FROM, ONBUILD, VOLUME, EXPOSE, ENTRYPOINT, CMD, COMMIT, PUSH are verboten 

293 failed = False 

294 if shutitfile_representation['shutitfile']['cmd'] != '' or shutitfile_representation['shutitfile']['volume'] != [] or shutitfile_representation['shutitfile']['onbuild'] != [] or shutitfile_representation['shutitfile']['expose'] != [] or shutitfile_representation['shutitfile']['entrypoint'] != []: 

295 failed = True 

296 for item in shutitfile_representation['shutitfile']['script']: 

297 if item[0] in ('PUSH','COMMIT'): 

298 failed = True 

299 break 

300 if failed: 

301 shutit.fail('One of FROM, ONBUILD, VOLUME, EXPOSE, ENTRYPOINT or CMD, COMMIT, PUSH used in ShutItFile not using the Docker delivery method.')

314 sections = {} 

315 sections.update({'header_section':'\n# Created from shutitfile: ' + skel_shutitfile + '\n# Maintainer: ' + shutitfile_representation['shutitfile']['maintainer'] + '\nfrom shutit_module import ShutItModule\n\nclass ' + skel_module_name + skel_module_modifier + '(ShutItModule):\n\n\tdef is_installed(self, shutit):\n\t\treturn False'}) 

316 

317 # config section - this must be done first, as it collates the config 

318 # items that can be referenced later 

319 config_section = '' 

320 if shutitfile_representation['shutitfile']['module_id']:

512def handle_shutitfile_config_line(line): 

513 shutitfile_command = line[0].upper() 

514 build = '' 

515 numtabs = 2 

516 assert shutitfile_command in ('CONFIG','CONFIG_SECRET'), '%r is not a handled config command' % shutitfile_command 

517 if shutitfile_command in ('CONFIG','CONFIG_SECRET'): 

518 shutitfile_args = parse_shutitfile_args(line[1]) 

519 assert isinstance(shutitfile_args, (dict,list)) 

520 if shutitfile_command == 'CONFIG': 

521 secret_str = 'False' 

522 elif shutitfile_command == 'CONFIG_SECRET': 

523 secret_str = 'True' 

524 if isinstance(shutitfile_args, list): 

525 assert len(shutitfile_args) in (1,2), '' 

526 cfg_name = shutitfile_args[0] 

527 if len(shutitfile_args) == 1: 

528 build += """\n""" + numtabs*'\t' + """shutit.get_config(self.module_id,'""" + cfg_name + """',secret=""" + secret_str + """)""" 

529 elif len(shutitfile_args) == 2: 

530 cfg_default = shutitfile_args[1] 

531 build += """\n""" + numtabs*'\t' + """shutit.get_config(self.module_id,'""" + cfg_name + """',default='""" + cfg_default + """',secret=""" + secret_str + """)""" 

532 return build

856def scan_text(text): 

857 """Scan text, and replace items that match shutit's pattern format, ie: 

858 {{ shutit.THING }} 

859 """ 

860 while True: 

861 match = re.match("(.*){{ shutit.(.*) }}(.*)$", text) 

862 if match: 

863 before = match.group(1) 

864 name = match.group(2) 

865 after = match.group(3) 

866 text = before + """''' + shutit.cfg[self.module_id][\"""" + name + """\"] + '''""" + after 

867 else: 

868 break 

869 return text

872# Get the section of the shutitfile we are in. 

873def shutitfile_get_section(shutitfile_command, current): 

874 match = re.match(r'^(.*)_(BEGIN|END)$',shutitfile_command) 

875 if match: 

876 section = match.group(1) 

877 stage = match.group(2) 

878 if stage == 'BEGIN': 

879 return section.lower() 

880 return 'build'
