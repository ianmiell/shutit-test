21def gather_module_paths(): 

22 import shutit_global 

23 shutit_global_object = shutit_global.shutit_global_object 

24 owd = shutit_global_object.owd 

25 shutit_module_paths = set() 

26 for shutit_object in shutit_global.shutit_global_object.shutit_objects: 

27 shutit_module_paths = shutit_module_paths.union(set(shutit_object.host['shutit_module_path'])) 

28 if '.' in shutit_module_paths: 

29 shutit_module_paths.remove('.') 

30 shutit_module_paths.add(owd)

139 """Simpler thread to track whether main thread has been quiet for long enough

142 import shutit_global 

143 last_msg = '' 

144 while True: 

145 printed_anything = False

147 this_msg = '' 

148 this_header = ''

150 # ignore own thread:
