457 def write_out_lines_to_fit_pane(self, pane, p_lines, title): 

458 assert pane is not None 

459 assert isinstance(pane, SessionPane) 

460 assert isinstance(title, unicode) 

461 pane_width = pane.get_width() 

462 pane_height = pane.get_height() 

463 assert pane_width > 39 

464 assert pane_height > 19 

465 # We reserve one row at the end as a pane status line 

466 available_pane_height = pane.get_height() - 1 

467 lines_in_pane_str_arr = [] 

468 p_lines_str = [] 

469 for session_pane_line in p_lines: 

470 assert isinstance(session_pane_line, SessionPaneLine) 

471 p_lines_str.append(session_pane_line.line_str) 

472 p_lines = p_lines_str 

473 p_lines_str = None 

474 # Scrub any ansi escape sequences. 

475 ansi_escape = re.compile(r'(\x9B|\x1B\[)[0-?]*[ -\/]*[@-~]') 

476 if not self.shutit_global.ispy3: 

477 lines = [ ansi_escape.sub('', line).strip().decode('utf-8') for line in p_lines ] 

478 else: 

479 lines = [ ansi_escape.sub('', line).strip() for line in p_lines ] 

480 # If the last line is blank we can just skip it. 

481 if len(lines) > 0 and lines[-1] == '': 

482 lines = lines[:-1] 

483 for line in lines: 

484 # Take the next line in the stream. If it's greater than the pane_width, 

485 # Then parcel over multiple lines 

486 while len(line) > pane_width-1 and len(line) > 0: 

487 lines_in_pane_str_arr.append(line[:pane_width-1]) 

488 line = line[pane_width-1:] 

489 lines_in_pane_str_arr.append(line) 

490 # Status line: 

491 lines_in_pane_str_arr.append(title) 

492 top_y = pane.top_left_y 

493 bottom_y = pane.bottom_right_y 

494 for i, line in zip(reversed(range(top_y,bottom_y)), reversed(lines_in_pane_str_arr)): 

495 # Status on bottom line 

496 # If this is on the top, and height + top_y value == i (ie this is the last line of the pane) 

497 # OR this is on the bottom (ie top_y is not 1), and height + top_y == i 

498 # One or both of these help prevent glitches on the screen. Don't know why. Maybe replace with more standard list TODO 

499 if (top_y == 1 and available_pane_height + top_y == i) or (top_y != 1 and available_pane_height + top_y == i): 

500 self.screen_arr[i:i+1, pane.top_left_x:pane.top_left_x+len(line)] = [cyan(invert(line))] 

501 else: 

502 self.screen_arr[i:i+1, pane.top_left_x:pane.top_left_x+len(line)] = [line]
